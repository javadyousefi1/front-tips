Unread Messages Counter

A practical example: simulate incoming messages and track unread count with a badge.

💬
0
Click "New Message" to simulate receiving a message...
let unread = 0;

function onNewMessage() {
  unread++;
  navigator.setAppBadge(unread);
}

function onMarkAllRead() {
  unread = 0;
  navigator.clearAppBadge();
}
